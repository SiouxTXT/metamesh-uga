### A.2.2 Modello di Dati per Agenti Paganti

**Tabella**: `agents` (D1)
```sql
CREATE TABLE agents (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  agent_id TEXT UNIQUE NOT NULL,           -- UUID dell'agente
  wallet_address TEXT UNIQUE NOT NULL,     -- Address Base per pagamenti
  email TEXT,
  name TEXT,
  plan TEXT DEFAULT 'free',                -- free | pay_as_you_go | enterprise
  budget_limit_usd DECIMAL(10,2) DEFAULT 0, -- Budget mensile (0 = illimitato)
  current_spent_usd DECIMAL(10,2) DEFAULT 0,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

**Tabella**: `agent_wallets` (D1)
```sql
CREATE TABLE agent_wallets (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  agent_id INTEGER NOT NULL,
  chain TEXT NOT NULL,                     -- 'base', 'ethereum', 'polygon'
  currency TEXT NOT NULL,                  -- 'USDC', 'USDT', 'DAI'
  balance DECIMAL(20,6) DEFAULT 0,         -- Saldo in valuta
  last_refreshed TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (agent_id) REFERENCES agents(id)
);
```

**Tabella**: `tool_pricing` (D1)
```sql
CREATE TABLE tool_pricing (
  tool_name TEXT PRIMARY KEY,
  price_per_call_usd DECIMAL(10,6) DEFAULT 0.001, -- $0.001 = 0.1 cent
  price_per_call_usdc DECIMAL(10,6) DEFAULT 0.001,
  x402_enabled BOOLEAN DEFAULT TRUE,
  discount_bulk BOOLEAN DEFAULT TRUE,      -- Sconti per volume
  min_call_volume INTEGER DEFAULT 0,       -- Soglia per sconti
  bulk_price DECIMAL(10,6) DEFAULT 0.0005,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

**Tabella**: `agent_usage` (D1 - caching, Analytics Engine per real-time)
```sql
CREATE TABLE agent_usage (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  agent_id INTEGER NOT NULL,
  tool_name TEXT NOT NULL,
  call_count INTEGER DEFAULT 0,
  total_spent_usd DECIMAL(10,6) DEFAULT 0,
  last_called TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (agent_id) REFERENCES agents(id)
);
```

### A.3.1 x402 Protocol (HTTP 402)

**Endpoint**: `POST /v1/agent/pay`

**Richiesta** (da agente AI):
```json
{
  "tool": "gmail_send_email",
  "params": { "to": "user@example.com", "subject": "Test" },
  "payment": {
    "signature": "0x...",           // Firma EIP-712
    "amount": "0.001",
    "currency": "USDC",
    "chain": "base",
    "payer": "0x123...",
    "nonce": "123456",
    "expiresAt": "2026-07-16T12:00:00Z"
  }
}
```

**Verifica Pagamento**:
```javascript
// packages/gateway/src/x402.js
import { verifyTypedData, recoverAddress } from 'viem';
import { base } from 'viem/chains';

async function verifyX402Payment(payment, env) {
  // 1. Verifica nonce (anti-replay)
  const used = await env.DB.prepare(
    "SELECT * FROM used_nonces WHERE nonce = ? AND agent_id = ?"
  ).bind(payment.nonce, payment.payer).first();
  if (used) throw new Error('Nonce already used');

  // 2. Verifica firma EIP-712
  const domain = {
    name: 'MetaMesh x402',
    version: '1',
    chainId: 8453, // Base
    verifyingContract: env.X402_CONTRACT_ADDRESS
  };
  
  const types = {
    Payment: [
      { name: 'payer', type: 'address' },
      { name: 'amount', type: 'uint256' },
      { name: 'currency', type: 'address' },
      { name: 'nonce', type: 'uint256' },
      { name: 'expiresAt', type: 'uint256' }
    ]
  };
  
  const recovered = recoverAddress({
    domain,
    types,
    primaryType: 'Payment',
    message: payment,
    signature: payment.signature
  });
  
  if (recovered !== payment.payer) {
    throw new Error('Invalid signature');
  }

  // 3. Verifica saldo
  const balance = await getAgentBalance(payment.payer, 'base', 'USDC');
  if (balance < payment.amount) {
    throw new Error('Insufficient balance');
  }

  // 4. Segna nonce come usato
  await env.DB.prepare(
    "INSERT INTO used_nonces (nonce, agent_id, used_at) VALUES (?, ?, datetime('now'))"
  ).bind(payment.nonce, payment.payer).run();

  // 5. Aggiorna wallet dell'agente
  await env.DB.prepare(
    "UPDATE agent_wallets SET balance = balance - ? WHERE agent_id = ? AND chain = 'base' AND currency = 'USDC'"
  ).bind(payment.amount, payment.payer).run();

  return { verified: true, payer: payment.payer };
}
```

```javascript
// packages/gateway/src/marketplace.js
export async function getAgentMarketplace(request, env) {
  const { agent_id } = request.query;
  
  // Recupera tool con pricing
  const tools = await env.DB.prepare(`
    SELECT 
      t.name, t.category, t.description, t.popularity_score,
      p.price_per_call_usd, p.price_per_call_usdc, p.x402_enabled,
      (SELECT AVG(latency_ms) FROM usage_log WHERE tool_name = t.name AND status = 'success' AND called_at > datetime('now', '-7 days')) as estimated_latency_ms
    FROM tools t
    JOIN tool_pricing p ON t.name = p.tool_name
    WHERE t.deprecated = FALSE AND p.x402_enabled = TRUE
    ORDER BY t.popularity_score DESC
    LIMIT 50
  `).all();
  
  // Recupera info agente
  const agent = await env.DB.prepare(
    "SELECT * FROM agents WHERE agent_id = ?"
  ).bind(agent_id).first();
  
  const wallet = await env.DB.prepare(
    "SELECT balance FROM agent_wallets WHERE agent_id = ? AND chain = 'base' AND currency = 'USDC'"
  ).bind(agent.id).first();
  
  return {
    tools: tools.results,
    total: tools.results.length,
    page: 1,
    agent_info: {
      agent_id: agent.agent_id,
      wallet_address: agent.wallet_address,
      balance_usdc: wallet.balance,
      budget_limit_usd: agent.budget_limit_usd,
      spent_this_month_usd: agent.current_spent_usd
    }
  };
}
```

### A.3.3 Usage Billing per Agent

**Cron Job** (`metamesh-agent-billing`):
```javascript
// packages/alerts/src/agent-billing.js
export default {
  async scheduled(event, env, ctx) {
    // 1. Genera fatture mensili per agenti
    await generateAgentInvoices(env);
    
    // 2. Controlla budget e invia alert
    await checkAgentBudgets(env);
    
    // 3. Aggiorna ranking dei tool per agenti
    await updateToolPopularity(env);
  }
};

async function generateAgentInvoices(env) {
  const agents = await env.DB.prepare(`
    SELECT * FROM agents WHERE plan != 'free' AND current_spent_usd > 0
  `).all();
  
  for (const agent of agents) {
    const invoice = {
      agent_id: agent.agent_id,
      period: 'monthly',
      total_usd: agent.current_spent_usd,
      tools: await getAgentToolUsage(agent.id, env),
      generated_at: new Date().toISOString()
    };
    
    // Genera PDF
    const pdf = await generateAgentInvoicePDF(invoice);
    
    // Invia email
    if (agent.email) {
      await sendEmail(
        agent.email,
        `📊 MetaMesh Agent Invoice - ${new Date().toLocaleMonthString()}`,
        pdf
      );
    }
    
    // Reset per il prossimo mese
    await env.DB.prepare(
      "UPDATE agents SET current_spent_usd = 0 WHERE id = ?"
    ).bind(agent.id).run();
  }
}

async function checkAgentBudgets(env) {
  const agents = await env.DB.prepare(`
    SELECT * FROM agents WHERE budget_limit_usd > 0
  `).all();
  
  for (const agent of agents) {
    const spend = await env.DB.prepare(`
      SELECT SUM(total_spent_usd) as total 
      FROM agent_usage 
      WHERE agent_id = ? AND last_called > datetime('now', '-1 month')
    `).bind(agent.id).first();
    
    const usagePercent = (spend.total / agent.budget_limit_usd) * 100;
    
    if (usagePercent > 80) {
      await sendAlert(
        agent.agent_id,
        `⚠️ Hai usato il ${usagePercent.toFixed(0)}% del tuo budget mensile (${spend.total}/${agent.budget_limit_usd})`
      );
    }
    
    if (usagePercent > 100) {
      // Blocca l'agente
      await env.DB.prepare(
        "UPDATE agents SET suspended = TRUE WHERE id = ?"
      ).bind(agent.id).run();
      await sendAlert(
        agent.agent_id,
        `🚫 Budget esaurito. Ricarica per continuare.`
      );
    }
  }
}
```

```javascript
// packages/gateway/src/agent-wallet.js
export async function topupAgentWallet(request, env) {
  const { agent_id, amount_usdc, payment_method, stripe_payment_intent_id } = await request.json();
  
  // Verifica pagamento Stripe
  if (payment_method === 'stripe') {
    const intent = await stripe.paymentIntents.retrieve(stripe_payment_intent_id);
    if (intent.status !== 'succeeded') {
      throw new Error('Payment not successful');
    }
  }
  
  // Aggiorna wallet su Base
  const tx = await sendUSDC(agent_id, amount_usdc);
  
  // Aggiorna D1
  await env.DB.prepare(`
    UPDATE agent_wallets 
    SET balance = balance + ? 
    WHERE agent_id = (SELECT id FROM agents WHERE agent_id = ?) 
    AND chain = 'base' AND currency = 'USDC'
  `).bind(amount_usdc, agent_id).run();
  
  // Log
  await env.ANALYTICS.writeDataPoint({
    blobs: [agent_id, "topup", "success"],
    doubles: [amount_usdc],
    indexes: ["agent_wallet"]
  });
  
  return {
    status: "success",
    new_balance_usdc: await getAgentBalance(agent_id, env),
    transaction_hash: tx.hash
  };
}
```
